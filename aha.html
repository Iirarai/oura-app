    async function loadOuraData() {

        const P = 'https://corsproxy.io/?url=',

            enc = encodeURIComponent;

        const [r1, r2] = await Promise.all([

            headers: h

        }),

        fetch(`${P}${enc(`https://api.ouraring.com/v2/usercollection/daily_readiness?start_date=${s}&end_date=${e}`)}`, {

            headers: h

        if (!r1.ok)

    function avg(arr) {

        return arr.length ? arr.reduce((a, b) => a + b, 0) / arr.length : 0;

    }

    function scoreColor(s) {

    function trendArrow(curr, prev) {

        if (!curr || !prev)

            return '';

        const d = curr - prev;

        if (Math.abs(d) < 2)

            return '<span class="kpi-trend trend-flat">→</span>';

        return d > 0 ? `<span class="kpi-trend trend-up">↑ +${Math.round(d)}</span>` : `<span class="kpi-trend trend-down">↓ ${Math.round(d)}</span>`;

    }

    async function callClaude(system, user, maxTokens=600) {

            body: JSON.stringify({

                model: 'claude-sonnet-4-20250514',

                max_tokens: maxTokens,

                system,

        });

    async function sendMessage() {

        if (!text || isThinking)

        const fmt = text.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>').replace(/\*(.*?)\*/g, '<em>$1</em>').split('\n\n').map(p => `<p>${p.replace(/\n/g, '<br>')}</p>`).join('');

        div.innerHTML = role === 'ai' ? `${av}<div class="msg-bubble">${fmt}</div>` : `${av}<div class="msg-bubble">${text}</div>`;

        el.appendChild(div);

        el.scrollTop = el.scrollHeight;

    }
